Rislozione di problemi relativi interfaccia grafica

# 1. Verifica che lightdm non sia bloccato e riavvialo con un ritardo
sudo systemctl stop lightdm
sleep 5
sudo systemctl start lightdm

# 2. Controlla e ripristina il file di configurazione di lightdm
sudo nano /etc/lightdm/lightdm.conf  # Verifica la configurazione
sudo rm /etc/lightdm/lightdm.conf    # Rimuovi la configurazione (se necessario)
sudo apt-get install --reinstall lightdm  # Reinstalla lightdm

# 3. Controlla i log di lightdm per trovare indizi sugli errori
cat /var/log/lightdm/lightdm.log

# 4. Imposta lightdm come display manager predefinito
sudo dpkg-reconfigure lightdm

# 5. Disabilita e riabilita il servizio lightdm
sudo systemctl disable lightdm
sudo systemctl enable lightdm

# 6. Verifica e reinstalla le dipendenze necessarie per il funzionamento di lightdm
sudo apt-get install --reinstall xserver-xorg-core xserver-xorg xinit

# 7. (Opzionale) Installa un altro display manager per testare se il problema è specifico di lightdm
sudo apt-get install gdm3

# 8. Aggiorna il sistema e ripara eventuali pacchetti danneggiati
sudo apt-get update
sudo apt-get upgrade
sudo apt-get --fix-broken install

# 9. Riavvia il sistema per applicare tutte le modifiche
sudo reboot
